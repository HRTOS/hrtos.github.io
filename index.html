任务调度核心（Preemptive Scheduler）

// ===============================
// RTOS 核心调度器（优先级抢占）
// ===============================

typedef struct
{
    uint8_t  priority;     // 优先级（0最高）
    uint8_t  state;        // 任务状态
    uint16_t *sp;          // 栈指针
} os_tcb_t;

extern os_tcb_t os_task_table[MAX_TASK];
extern uint8_t  current_task;
extern uint8_t  ready_bitmap;

// 查找最高优先级就绪任务（位图优化）
static uint8_t os_find_highest_ready(void)
{
    uint8_t i;

    for(i = 0; i < MAX_TASK; i++)
    {
        if(ready_bitmap & (1 << i))
        {
            return i;
        }
    }

    return IDLE_TASK;
}

// 核心调度入口
void os_schedule(void)
{
    uint8_t next;

    DISABLE_INT();

    // 获取最高优先级任务
    next = os_find_highest_ready();

    // 判断是否需要切换
    if(next != current_task)
    {
        // 保存当前任务上下文
        os_save_context(current_task);

        // 更新当前任务
        current_task = next;

        // 恢复目标任务上下文
        os_restore_context(current_task);
    }

    ENABLE_INT();
}

// 调度触发（用于中断或系统调用）
void os_schedule_request(void)
{
    if(!in_interrupt)
    {
        os_schedule();
    }
    else
    {
        // 延迟到中断退出再调度
        need_schedule = 1;
    }
}

任务生命周期管理（Task Lifecycle）

#include <hrtos.h>
#include <reg52.h>

/* ===== 任务控制块示意 ===== */
typedef struct {
    void (*entry)(void);
    u8 priority;
    u8 state;
    u16 *sp;
} os_tcb_t;

/* ===== 任务函数 ===== */
void task_led()
{
    while(1)
    {
        P0 ^= 0xFF;
        os_delay(10);
    }
}

void task_monitor()
{
    while(1)
    {
        /* 系统状态监控 */
        if(os_task_state(1) == OS_BLOCKED)
        {
            os_task_ready(1);
        }
        os_delay(5);
    }
}

/* ===== 初始化入口 ===== */
void hrtos_main()
{
    EA = 0;

    /* 创建任务（入口 + 优先级 + 栈大小 + ID） */
    os_task(task_led,     1,  8, 1);
    os_task(task_monitor, 2,  6, 2);

    /* 挂起低优先级任务 */
    os_task_hang(2);

    EA = 1;

    while(1)
    {
        /* 主循环空转，调度由内核控制 */
    }
}

中断调度与上下文切换（Interrupt Scheduling）

#include <hrtos.h>
#include <reg52.h>

/* ===== 当前运行任务ID ===== */
volatile u8 current_task = 0;

/* ===== 保存上下文 ===== */
void os_save_context()
{
    OS_SP[current_task] = SP;   // 保存栈指针
}

/* ===== 恢复上下文 ===== */
void os_restore_context(u8 next)
{
    SP = OS_SP[next];           // 恢复目标任务栈
    current_task = next;
}

/* ===== 调度器核心 ===== */
u8 os_schedule()
{
    u8 next = current_task;

    /* 简化优先级选择 */
    next = os_select(next);

    return next;
}

/* ===== 中断入口 ===== */
void os_interrupt_entry()
{
    u8 next;

    os_save_context();          // 保存当前任务

    next = os_schedule();       // 选择下一个任务

    if(next != current_task)
    {
        os_restore_context(next);  // 切换任务
    }
}

/* ===== 定时器中断（调度触发） ===== */
void timer0_isr() interrupt 1
{
    TF0 = 0;

    os_interrupt_entry();       // 进入内核调度
}

/* ===== 外部中断示例 ===== */
void ext0_isr() interrupt 0
{
    /* 处理外设事件 */
    os_interrupt_entry();       // 可触发任务切换
}

/* ===== 系统启动 ===== */
void hrtos_main()
{
    EA = 0;

    os_task(task1, 1, 8, 1);
    os_task(task2, 2, 8, 2);

    EA = 1;

    while(1)
    {
        /* 空循环，调度由中断驱动 */
    }
}
