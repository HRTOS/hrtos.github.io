// RTOS 核心调度器（优先级抢占）
void os_task_schedule(void)
{
    uint8_t next;

    DISABLE_INT();
    next = os_find_highest_ready();

    if(next != current_task)
    {
        os_save_context(current_task);
        current_task = next;
        os_restore_context(current_task);
    }

    ENABLE_INT();
}
